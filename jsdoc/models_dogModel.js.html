/**
 * Dog Model
 * @module models/dogModel
 * @requires mongoose
 */
const mongoose = require('mongoose')

/**
 * Dog Information Schema 
 * @type {dog_id:String, title:String, age:Number, description:String, content:String, images:Object, category:String, checked:Boolean}
*/
const dogSchema = new mongoose.Schema({
    dog_id:{
        type: String,
        unique: true,
        trim: true,
        required: true
    },
    title:{
        type: String,
        trim: true,
        required: true
    },
    age:{
        type: Number,
        trim: true,
        required: true
    },
    description:{
        type: String,
        required: true
    },
    content:{
        type: String,
        required: true
    },
    images:{
        type: Object,
        required: true
    },
    category:{
        type: String,
        required: true
    },
    checked:{
        type: Boolean,
        default: false
    }
}, {
    timestamps: true //important
})


module.exports = mongoose.model("dogs", dogSchema)
