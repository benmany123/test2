const mongoose = require('mongoose')

/**Breed Schema */
const categorySchema = new mongoose.Schema({
    /**
     * @type {string}
     */
    name: {
        type: String,
        required: true,
        trim: true,
        unique: true
    }
}, {
    timestamps: true
})

module.exports = mongoose.model("Category", categorySchema)
